## Install

```bash
npm install --save oum-react-modal
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'oum-react-modal'
import 'oum-react-modal/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
